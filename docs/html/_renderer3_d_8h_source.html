1791 TGX_INLINE inline bool _discardMeshlet16b(const fVec3& sphere_center, float sphere_radius, const fVec3& cone_dir, float cone_cos) const
1792 {
1793 if (cone_cos <= -1.0f) return false;
1794
1795 const fVec4 D = _r_modelViewM.mult0(cone_dir);
1796 const float dd = D.x * D.x + D.y * D.y + D.z * D.z;
1797 if (dd <= 1.0e-20f) return false;
1798
1799 float dot;
1800 float len2;
1801 if (_ortho)
1802 {
1803 dot = D.z; // object-to-camera direction is +Z in view space.
1804 len2 = dd;
1805 }
1806 else
1807 {
1808 const fVec3 anchor = sphere_center - (cone_dir * sphere_radius);
1809 const fVec4 A = _r_modelViewM.mult1(anchor);
1810 dot = -(D.x * A.x + D.y * A.y + D.z * A.z);
1811 const float aa = A.x * A.x + A.y * A.y + A.z * A.z;
1812 if (aa <= 1.0e-20f) return false;
1813 len2 = dd * aa;
1814 }
1815
1816 const float c2len2 = cone_cos * cone_cos * len2;
1817 const float dot2 = dot * dot;
1818 return (cone_cos >= 0.0f) ? ((dot < 0.0f) || (dot2 < c2len2))
1819 : ((dot < 0.0f) && (dot2 > c2len2));
1820 }

1866 template<bool CHECKRANGE, bool USE_BLENDING> TGX_INLINE inline void drawPixelZbuf(int x, int y, color_t color, float opacity, float z)
1867 {
1868 if (CHECKRANGE && ((x < 0) || (x >= _uni.im->lx()) || (y < 0) || (y >= _uni.im->ly()))) return;
1869 ZBUFFER_t& W = _uni.zbuf[x + _uni.im->lx() * y];
1870 const ZBUFFER_t aa = (std::is_same<ZBUFFER_t, uint16_t>::value) ? ((ZBUFFER_t)(z * _uni.wa + _uni.wb)) : ((ZBUFFER_t)z);
1871 if (W < aa)
