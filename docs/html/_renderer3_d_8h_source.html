2705 inline bool _discardMeshlet16b(const fVec3& sphere_center, float sphere_radius, const fVec3& cone_dir, float cone_cos) const
2706 {

2710 const float dd = D.x * D.x + D.y * D.y + D.z * D.z;
2711 if (dd <= 1.0e-20f) return false;

2714 float len2;
