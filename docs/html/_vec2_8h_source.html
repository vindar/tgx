598 template<typename T> TGX_INLINE inline bool Vec2<T>::setAsIntersection(const Vec2<T> & LA1, const Vec2<T> & LA2, const Vec2<T> & LB1, const Vec2<T> & LB2)
599 {
600 const T a1 = LA2.y - LA1.y;
601 const T b1 = LA1.x - LA2.x;
602 const T a2 = LB2.y - LB1.y;
603 const T b2 = LB1.x - LB2.x;
604 const T delta = a1 * b2 - a2 * b1;
605 if (delta == 0) { return false; }
606 const T c1 = LA1.x * a1 + LA1.y * b1;
607 const T c2 = LB1.x * a2 + LB1.y * b2;
608 x = ((b1 == 0) ? LA1.x : ((b2 == 0) ? LB1.x : (T)((b2 * c1 - b1 * c2) / delta))); // complicated but insures perfect clipping
609 y = ((a1 == 0) ? LA1.y : ((a2 == 0) ? LB1.y : (T)((a1 * c2 - a2 * c1) / delta))); // for horizontal and vertical lines
610 return true;
611 }
