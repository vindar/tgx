45#define TGX_RASTERIZE_SUBPIXEL256 (1 << TGX_RASTERIZE_SUBPIXEL_BITS)
46#define TGX_RASTERIZE_SUBPIXEL128 (1 << (TGX_RASTERIZE_SUBPIXEL_BITS -1))
47#define TGX_RASTERIZE_MULT256(X) ((X) << (TGX_RASTERIZE_SUBPIXEL_BITS))
48#define TGX_RASTERIZE_MULT128(X) ((X) << (TGX_RASTERIZE_SUBPIXEL_BITS -1))
49#define TGX_RASTERIZE_DIV256(X) ((X) >> (TGX_RASTERIZE_SUBPIXEL_BITS))

104 template<typename SHADER_FUNCTION, typename RASTERIZER_PARAMS>
105 void rasterizeTriangle(const int LX, const int LY, const RasterizerVec4 & V0, const RasterizerVec4 & V1, const RasterizerVec4 & V2, const int32_t offset_x, const int32_t offset_y, const RASTERIZER_PARAMS & data, SHADER_FUNCTION shader_fun)
106 {
107
108 // assuming that clipping was already perfomed and that V0, V1, V2 are in a reasonable "range" so no overflow will occur.
109 const float mx = (float)(TGX_RASTERIZE_MULT128(LX));
110 const float my = (float)(TGX_RASTERIZE_MULT128(LY));
111 const iVec2 P0((int32_t)floorf(V0.x * mx), (int32_t)floorf(V0.y * my));
112 const iVec2 sP1((int32_t)floorf(V1.x * mx), (int32_t)floorf(V1.y * my));
113 const iVec2 sP2((int32_t)floorf(V2.x * mx), (int32_t)floorf(V2.y * my));
114
115 const int32_t umx = min(min(P0.x, sP1.x), sP2.x);
116 const int32_t uMx = max(max(P0.x, sP1.x), sP2.x);
117 const int32_t umy = min(min(P0.y, sP1.y), sP2.y);
118 const int32_t uMy = max(max(P0.y, sP1.y), sP2.y);
119
120 const bool c32 = ((uMx - umx < 32768) && (uMy - umy < 32768));
121 int32_t a;
122 if (c32)
123 { // 32 bits computation
124 a = (((sP2.x - P0.x) * (sP1.y - P0.y)) - ((sP2.y - P0.y) * (sP1.x - P0.x))); // aera
125 if (a == 0) return; // do not draw flat triangles
126 }
127 else
128 { // 64 bits computations
129 int64_t a64 = (((int64_t)(sP2.x - P0.x)) * ((int64_t)(sP1.y - P0.y))) - (((int64_t)(sP2.y - P0.y)) * ((int64_t)(sP1.x - P0.x))); // aera
130 if (a64 == 0) return; // do not draw flat triangles
131 a = (a64 > 0) ? 1 : -1; // real aera value does not matter, onlythe sign
132 }
133
134 int32_t xmin = (umx + TGX_RASTERIZE_MULT128(LX)) / TGX_RASTERIZE_SUBPIXEL256; // use division and not bitshift
135 int32_t xmax = (uMx + TGX_RASTERIZE_MULT128(LX)) / TGX_RASTERIZE_SUBPIXEL256; // in case values are negative.
136 int32_t ymin = (umy + TGX_RASTERIZE_MULT128(LY)) / TGX_RASTERIZE_SUBPIXEL256; //
137 int32_t ymax = (uMy + TGX_RASTERIZE_MULT128(LY)) / TGX_RASTERIZE_SUBPIXEL256; //
138
139 // intersect the sub-image with the triangle bounding box.
140 int32_t sx = data.im->lx();
141 int32_t sy = data.im->ly();
142 int32_t ox = offset_x;
143 int32_t oy = offset_y;
144 if (ox < xmin) { sx -= (xmin - ox); ox = xmin; }
145 if (ox + sx > xmax) { sx = xmax - ox + 1; }
146 if (sx <= 0) return;
147 if (oy < ymin) { sy -= (ymin - oy); oy = ymin; }
148 if (oy + sy > ymax) { sy = ymax - oy + 1; }
149 if (sy <= 0) return;
150
151 const RasterizerVec4& fP1 = (a > 0) ? V1 : V2;
152 const RasterizerVec4& fP2 = (a > 0) ? V2 : V1;
153 const iVec2& P1 = (a > 0) ? sP1 : sP2;
154 const iVec2& P2 = (a > 0) ? sP2 : sP1;
155
156 const int32_t us = TGX_RASTERIZE_MULT256(ox) - TGX_RASTERIZE_MULT128(LX) + TGX_RASTERIZE_SUBPIXEL128; // start pixel position
157 const int32_t vs = TGX_RASTERIZE_MULT256(oy) - TGX_RASTERIZE_MULT128(LY) + TGX_RASTERIZE_SUBPIXEL128; //
158
159 ox -= offset_x;
160 oy -= offset_y;
161
162 int32_t dx1 = P1.y - P0.y;
163 int32_t dy1 = P0.x - P1.x;
164 int32_t dx2 = P2.y - P1.y;
165 int32_t dy2 = P1.x - P2.x;
166 int32_t dx3 = P0.y - P2.y;
167 int32_t dy3 = P2.x - P0.x;
168
169 int32_t O1, O2, O3;
170
171 if (c32)
172 { // 32 bits computation
173 O1 = ((us - P0.x) * dx1) + ((vs - P0.y) * dy1);
174 if ((dx1 < 0) || ((dx1 == 0) && (dy1 < 0))) O1--; // top left rule (beware, changes total aera).
175
176 O2 = ((us - P1.x) * dx2) + ((vs - P1.y) * dy2);
177 if ((dx2 < 0) || ((dx2 == 0) && (dy2 < 0))) O2--; // top left rule (beware, changes total aera).
178
179 O3 = ((us - P2.x) * dx3) + ((vs - P2.y) * dy3);
180 if ((dx3 < 0) || ((dx3 == 0) && (dy3 < 0))) O3--; // top left rule (beware, changes total aera).
181
182 dx1 *= (TGX_RASTERIZE_SUBPIXEL256);
183 dy1 *= (TGX_RASTERIZE_SUBPIXEL256);
184 dx2 *= (TGX_RASTERIZE_SUBPIXEL256);
185 dy2 *= (TGX_RASTERIZE_SUBPIXEL256);
186 dx3 *= (TGX_RASTERIZE_SUBPIXEL256);
187 dy3 *= (TGX_RASTERIZE_SUBPIXEL256);
188 }
189 else
190 { // 64 bits computation
191 int64_t dO1 = (((int64_t)(us - P0.x)) * ((int64_t)dx1)) + (((int64_t)(vs - P0.y)) * ((int64_t)dy1));
192 if ((dx1 < 0) || ((dx1 == 0) && (dy1 < 0))) dO1--; // top left rule (beware, changes total aera).
193 O1 = (dO1 >= 0) ? ((int32_t)TGX_RASTERIZE_DIV256(dO1)) : -((int32_t)TGX_RASTERIZE_DIV256(-dO1 + (TGX_RASTERIZE_SUBPIXEL256 - 1)));
194
195 int64_t dO2 = (((int64_t)(us - P1.x)) * ((int64_t)dx2)) + (((int64_t)(vs - P1.y)) * ((int64_t)dy2));
196 if ((dx2 < 0) || ((dx2 == 0) && (dy2 < 0))) dO2--; // top left rule (beware, changes total aera).
197 O2 = (dO2 >= 0) ? ((int32_t)TGX_RASTERIZE_DIV256(dO2)) : -((int32_t)TGX_RASTERIZE_DIV256(-dO2 + (TGX_RASTERIZE_SUBPIXEL256 - 1)));
198
199 int64_t dO3 = (((int64_t)(us - P2.x)) * ((int64_t)dx3)) + (((int64_t)(vs - P2.y)) * ((int64_t)dy3));
200 if ((dx3 < 0) || ((dx3 == 0) && (dy3 < 0))) dO3--; // top left rule (beware, changes total aera).
201 O3 = (dO3 >= 0) ? ((int32_t)TGX_RASTERIZE_DIV256(dO3)) : -((int32_t)TGX_RASTERIZE_DIV256(-dO3 + (TGX_RASTERIZE_SUBPIXEL256 - 1)));
202 }
203
204 // beware that O1 + O2 + O3 = 0 is possible now (but still should not discard the triangle)
205 // must be dealt with inside the shader
206
207 if (sx == 1)
208 {
209 while (((O1 | O2 | O3) < 0) && (sy > 0))
210 {
211 sy--;
212 oy++;
213 O1 += dy1;
214 O2 += dy2;
215 O3 += dy3;
216 }
217 if (sy == 0) return;
218 }
219 else if (sy == 1)
220 {
221 while (((O1 | O2 | O3) < 0) && (sx > 0))
222 {
223 sx--;
224 ox++;
225 O1 += dx1;
226 O2 += dx2;
227 O3 += dx3;
228 }
229 if (sx == 0) return;
230 }
231
232
233 if (dx1 > 0)
234 {
235 shader_fun(ox + (data.im->stride() * oy), sx, sy,
236 dx1, dy1, O1, fP2,
237 dx2, dy2, O2, V0,
238 dx3, dy3, O3, fP1,
239 data);
240 }
241 else if (dx2 > 0)
242 {
243 shader_fun(ox + (data.im->stride() * oy), sx, sy,
244 dx2, dy2, O2, V0,
245 dx3, dy3, O3, fP1,
246 dx1, dy1, O1, fP2,
247 data);
248 }
249 else
250 {
251 shader_fun(ox + (data.im->stride() * oy), sx, sy,
252 dx3, dy3, O3, fP1,
253 dx1, dy1, O1, fP2,
254 dx2, dy2, O2, V0,
255 data);
256 }
257 return;
258 }
